# Description

Implement an arbitrary-precision `Decimal` class.

Floating point numbers are the most common representation of non-integer real numbers in computing, and they're a common standard defined by [IEEE 754](https://en.wikipedia.org/wiki/IEEE_754). They're very flexible and versatile, but they do have some limitations. Famously, in floating point arithmetic, [`0.1 + 0.2 != 0.3`](http://0.30000000000000004.com/).

The solution to this issue is to find another, lossless way to model arbitrary-precision non-integer reals. This may be less efficient in terms of memory or processing speed than floating point numbers; the goal is to provide exact results.

Despite `Decimal` being a custom type, we should still be able to treat them as numbers: the `==`, `<`, `>`, `+`, `-`, and `*` operators should all work as expected on Decimals. For expediency, you are not required to implement division, as arbitrary-precision division can very quickly get out of hand. (How do you represent arbitrary-precision `1/3`?)

In Rust, the way to get these operations on custom types is to implement the relevant traits for your custom object. In particular, you'll need to implement at least `PartialEq`, `PartialOrd`, `Add`, `Sub`, and `Mul`. Strictly speaking, given that the decimal numbers form a total ordering, you should also implement `Eq` and `Ord`, though those traits are not checked for by these tests.

use std::cmp::Ordering;
use std::ops::{Add, Mul, Sub};

/// Type implementing arbitrary-precision decimal arithmetic.
///
/// The value is modeled as `sign * magnitude / 10^scale`, where `magnitude`
/// is an arbitrary-precision non-negative integer stored as base-10 digits in
/// little-endian order (`digits[0]` is the least significant digit). This keeps
/// every value exact regardless of how many integer or fractional digits it has.
#[derive(Debug, Clone)]
pub struct Decimal {
    /// `1` for positive values, `-1` for negative values. Always `1` for zero.
    sign: i8,
    /// Magnitude as base-10 digits, little-endian, with no high-order zeros.
    /// An empty vector represents zero.
    digits: Vec<u8>,
    /// Number of `digits` that fall after the decimal point.
    scale: usize,
}

impl Decimal {
    pub fn try_from(input: &str) -> Option<Decimal> {
        let mut chars = input.chars();
        let mut sign = 1i8;

        let mut rest = input;
        if let Some(first) = chars.next() {
            match first {
                '+' => rest = &input[1..],
                '-' => {
                    sign = -1;
                    rest = &input[1..];
                }
                _ => {}
            }
        }

        if rest.is_empty() {
            return None;
        }

        let mut parts = rest.split('.');
        let int_part = parts.next().unwrap_or("");
        let frac_part = parts.next().unwrap_or("");
        if parts.next().is_some() {
            return None; // more than one decimal point
        }

        if int_part.is_empty() && frac_part.is_empty() {
            return None;
        }
        if !int_part.bytes().all(|b| b.is_ascii_digit())
            || !frac_part.bytes().all(|b| b.is_ascii_digit())
        {
            return None;
        }

        // Build the magnitude (most significant first) then reverse to
        // little-endian for arithmetic.
        let digits: Vec<u8> = int_part
            .bytes()
            .chain(frac_part.bytes())
            .rev()
            .map(|b| b - b'0')
            .collect();

        let mut decimal = Decimal {
            sign,
            digits,
            scale: frac_part.len(),
        };
        decimal.normalize();
        Some(decimal)
    }

    /// Collapse to a canonical form: strip insignificant fractional and
    /// leading zeros so that equal values share an identical representation.
    fn normalize(&mut self) {
        while self.scale > 0 && self.digits.first() == Some(&0) {
            self.digits.remove(0);
            self.scale -= 1;
        }
        while self.digits.last() == Some(&0) {
            self.digits.pop();
        }
        if self.digits.is_empty() {
            self.sign = 1;
            self.scale = 0;
        }
    }

    /// Return the magnitude shifted so it represents the value at `target_scale`
    /// (which must be >= `self.scale`).
    fn aligned_magnitude(&self, target_scale: usize) -> Vec<u8> {
        let shift = target_scale - self.scale;
        let mut aligned = vec![0u8; shift];
        aligned.extend_from_slice(&self.digits);
        aligned
    }

    fn is_zero(&self) -> bool {
        self.digits.is_empty()
    }

    fn negated(&self) -> Decimal {
        let mut result = self.clone();
        if !result.is_zero() {
            result.sign = -result.sign;
        }
        result
    }
}

/// Compare two little-endian magnitudes, ignoring any high-order zeros.
fn cmp_magnitude(a: &[u8], b: &[u8]) -> Ordering {
    let len_a = a.iter().rposition(|&d| d != 0).map_or(0, |i| i + 1);
    let len_b = b.iter().rposition(|&d| d != 0).map_or(0, |i| i + 1);
    if len_a != len_b {
        return len_a.cmp(&len_b);
    }
    for i in (0..len_a).rev() {
        if a[i] != b[i] {
            return a[i].cmp(&b[i]);
        }
    }
    Ordering::Equal
}

/// Add two little-endian magnitudes.
fn add_magnitude(a: &[u8], b: &[u8]) -> Vec<u8> {
    let n = a.len().max(b.len());
    let mut out = Vec::with_capacity(n + 1);
    let mut carry = 0u8;
    for i in 0..n {
        let sum = a.get(i).copied().unwrap_or(0) + b.get(i).copied().unwrap_or(0) + carry;
        out.push(sum % 10);
        carry = sum / 10;
    }
    if carry > 0 {
        out.push(carry);
    }
    out
}

/// Subtract little-endian magnitude `b` from `a`. Assumes `a >= b`.
fn sub_magnitude(a: &[u8], b: &[u8]) -> Vec<u8> {
    let mut out = Vec::with_capacity(a.len());
    let mut borrow = 0i16;
    for (i, &da) in a.iter().enumerate() {
        let mut diff = da as i16 - b.get(i).copied().unwrap_or(0) as i16 - borrow;
        if diff < 0 {
            diff += 10;
            borrow = 1;
        } else {
            borrow = 0;
        }
        out.push(diff as u8);
    }
    out
}

/// Multiply two little-endian magnitudes.
fn mul_magnitude(a: &[u8], b: &[u8]) -> Vec<u8> {
    if a.is_empty() || b.is_empty() {
        return Vec::new();
    }
    let mut acc = vec![0u32; a.len() + b.len()];
    for (i, &da) in a.iter().enumerate() {
        for (j, &db) in b.iter().enumerate() {
            acc[i + j] += da as u32 * db as u32;
        }
    }
    let mut out = Vec::with_capacity(acc.len() + 1);
    let mut carry = 0u32;
    for value in acc {
        let cur = value + carry;
        out.push((cur % 10) as u8);
        carry = cur / 10;
    }
    while carry > 0 {
        out.push((carry % 10) as u8);
        carry /= 10;
    }
    out
}

impl PartialEq for Decimal {
    fn eq(&self, other: &Self) -> bool {
        self.cmp(other) == Ordering::Equal
    }
}

impl Eq for Decimal {}

impl PartialOrd for Decimal {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

impl Ord for Decimal {
    fn cmp(&self, other: &Self) -> Ordering {
        if self.sign != other.sign {
            return self.sign.cmp(&other.sign);
        }
        let common_scale = self.scale.max(other.scale);
        let a = self.aligned_magnitude(common_scale);
        let b = other.aligned_magnitude(common_scale);
        let magnitude_order = cmp_magnitude(&a, &b);
        if self.sign == 1 {
            magnitude_order
        } else {
            magnitude_order.reverse()
        }
    }
}

impl Add for Decimal {
    type Output = Decimal;

    fn add(self, other: Decimal) -> Decimal {
        let common_scale = self.scale.max(other.scale);
        let a = self.aligned_magnitude(common_scale);
        let b = other.aligned_magnitude(common_scale);

        let (sign, digits) = if self.sign == other.sign {
            (self.sign, add_magnitude(&a, &b))
        } else {
            match cmp_magnitude(&a, &b) {
                Ordering::Equal => (1, Vec::new()),
                Ordering::Greater => (self.sign, sub_magnitude(&a, &b)),
                Ordering::Less => (other.sign, sub_magnitude(&b, &a)),
            }
        };

        let mut result = Decimal {
            sign,
            digits,
            scale: common_scale,
        };
        result.normalize();
        result
    }
}

impl Sub for Decimal {
    type Output = Decimal;

    // Subtraction is addition of the negation: `a - b = a + (-b)`.
    #[allow(clippy::suspicious_arithmetic_impl)]
    fn sub(self, other: Decimal) -> Decimal {
        self + other.negated()
    }
}

impl Mul for Decimal {
    type Output = Decimal;

    fn mul(self, other: Decimal) -> Decimal {
        let mut result = Decimal {
            sign: self.sign * other.sign,
            digits: mul_magnitude(&self.digits, &other.digits),
            scale: self.scale + other.scale,
        };
        result.normalize();
        result
    }
}
